Demo1-ControllerAs给控制器取别名示例

var myApp = angular.module('myApp', []); //注意app创建时需要添加第二个参数
    myApp.controller('MainController',function(){
        var vm=this;
        vm.user={name: '时刻知(ShiKeZhi)',
                 website:'http://www.shikezhi.com',
                 turtorial:'https://github.com/johnpapa/angular-styleguide#controlleras-controller-syntax'
        };
    });

输出：{{main.user.name}} say, welcome to visit {{main.user.website}}

参考： {{main.user.turtorial}}
